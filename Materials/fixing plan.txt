PRODUCT_SHELF_LIFE_PARAMS = {
    PROD_001 {  # Fresh Fish
        shelf_life_base_days 1.5,
        temp_reference_C 4.0,     # Nhiệt độ tham chiếu (bảo quản lạnh)
        activation_energy_Ea 0.85, # Hệ số Arrhenius (từ literature)
        ambient_temp_monsoon_C 30, # Nhiệt độ môi trường mùa mưa Đà Nẵng
    },
    PROD_007 {  # Chicken
        shelf_life_base_days 2.0,
        temp_reference_C 4.0,
        activation_energy_Ea 0.78,
        ambient_temp_monsoon_C 30,
    },
    PROD_004 {  # Leafy Greens — ít nhạy cảm hơn

        ambient_temp_monsoon_C 30,
    }
}

def compute_spoilage_rate(product_id, transport_time_h, is_refrigerated, weather_scenario)
    
    Simplified Arrhenius model for spoilage during transport
    Rate doubles roughly every 10°C (Q10 rule)
    
    params = PRODUCT_SHELF_LIFE_PARAMS[product_id]
    T_eff = params[temp_reference_C] if is_refrigerated else params[ambient_temp_monsoon_C]
    T_ref = params[temp_reference_C]
    Q10 = 2.0  # Standard food science value
    rate_multiplier = Q10  ((T_eff - T_ref)  10.0)
    base_rate = 1.0  (params[shelf_life_base_days]  24)  # per hour
    spoilage_fraction = min(1.0, base_rate  rate_multiplier  transport_time_h)
    return spoilage_fraction
